## Wage Law Update Schedule

```javascript
class WageLawDatabase {
  constructor() {
    this.updateSchedule = 'MONTHLY';
    this.sources = ['Official government sources', 'ILO database', 'Legal partners'];
    this.countries = 200;
  }

  applyCorrectWageLaw(workerId, payrollEntry, payPeriod) {
    const worker = getWorker(workerId);
    const jurisdiction = worker.workLocation.jurisdiction;
    const applicableLaw = this.getApplicableLaw(jurisdiction, payPeriod.date);
    
    const minimumRequired = applicableLaw.calculateMinimum(
      payrollEntry.hoursWorked,
      payrollEntry.jobType,
      worker.employmentType
    );
    
    if (payrollEntry.grossPay < minimumRequired) {
      const correction = minimumRequired - payrollEntry.grossPay;
      wageTheftDetected(workerId, payrollEntry, correction, applicableLaw);
      return { corrected: true, correctionAmount: correction, law: applicableLaw.citation };
    }
    
    return { compliant: true, minimumRequired, paid: payrollEntry.grossPay };
  }
}
```
